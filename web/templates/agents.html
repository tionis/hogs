Quick Start
# 1. Download hogs-agent binary
wget https://github.com/tionis/hogs/releases/latest/download/hogs-agent -O /usr/local/bin/hogs-agent
chmod +x /usr/local/bin/hogs-agent

# 2. Create systemd unit (see above)
# 3. Start the agent
systemctl enable --now hogs-agent

# 4. Check status
systemctl status hogs-agent
journalctl -u hogs-agent -f
